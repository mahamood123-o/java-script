let name = prompt("Enter student name:"); let marks = []; for (let i = 1; i <= 5; i++) { let mark = parseInt(prompt(`Enter mark for subject ${i}:`)); marks.push(mark); } if (marks.some(isNaN)) { console.log("Invalid input. Please enter only numbers for marks."); } else { let total = marks.reduce((sum, val) => sum + val, 0); let average = total / 5; let result = ""; let grade = ""; if (marks.some(mark => mark < 35)) { result = "FAIL"; grade = "No Grade"; } else { result = "PASS"; if (average >= 75 && average <= 100) { grade = "A"; } else if (average >= 65 && average <= 74) { grade = "B"; } else if (average >= 45 && average <= 64) { grade = "C"; } else if (average >= 36 && average <= 44) { grade = "D"; } else { grade = "FAIL"; } } console.log("Student Name: " + name); console.log("Marks: " + marks.join(", ")); console.log("Average: " + average.toFixed(2)); console.log("Result: " + result); console.log("Grade: " + grade); }
